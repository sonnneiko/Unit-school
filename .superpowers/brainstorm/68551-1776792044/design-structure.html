Два независимых дерева маршрутов

Сейчас (App.tsx)
PrivateRoute
  AppLayout ← Sidebar (Главная, Курсы…)
    / → Dashboard
    /courses → Placeholder
    /progress → Placeholder
    /profile → Placeholder
    AdminRoute
      AdminLayout ← только breadcrumbs
        /admin/users
        /admin/courses
После (App.tsx)
PrivateRoute
  AppLayout ← Sidebar (Главная, Курсы…)
    / → Dashboard
    /courses → Placeholder
    /progress → Placeholder
    /profile → Placeholder
  AdminRoute
    AdminAppLayout ← AdminSidebar (новый)
      /admin → redirect
      /admin/dashboard → AdminDashboardPage (новая)
      /admin/users → UsersListPage
      /admin/users/new → NewUserPage
      /admin/users/:id → UserDetailPage*
      /admin/courses → CoursesListPage
      /admin/courses/:id → CourseDetailPage
      /admin/progress → AdminProgressPage (новая)

* UserDetailPage — убираем блок прогресса по курсам
